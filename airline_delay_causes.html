# As per the definion of Bureau of Transportation Statistics if the flight arrive later than 15 minutes then it will be marked as delay
# DelayStatus = 0, defines there was no delay  
# DelayStatus = 1, means slightly delayed
# DelayStatus = 2, means highly delayed
# ArrDelay is the sum of the following delays CarrierDelay, WeatherDelay, NASDelay, SecurityDelay and LateAircraftDelay 	
# airline_delay=
airline_delay_spark=airline_delay_spark.withColumn("date", expr("make_date(Year, Month, DayofMonth)"))
airline_delay_spark.registerTempTable("airline_delay_spark_temp")
airline_delay=spark.sql("""select CRSDepTime,UniqueCarrier,  AirTime , ArrDelay,Origin, 	Dest, 	Distance, TaxiIn, 	TaxiOut, Diverted,Cancelled,
                          CarrierDelay, 	WeatherDelay, 	NASDelay, 	SecurityDelay, 	LateAircraftDelay, 	date
                          ,case when arrdelay <= 15 then 1 when arrdelay  between 16 and 60 then 2
                          when arrdelay > 60 then 3  end DelayStatus, 
                          case when CancellationCode ='A' then 1 when CancellationCode ='B' then 2
                           when CancellationCode ='C' then 3 else 4 end CancellationCode   from airline_delay_spark_temp """)
